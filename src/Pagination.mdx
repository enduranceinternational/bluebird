import {
  Pager, Pagination,
} from 'react-bootstrap';
import { Playground } from 'docz';

# Pagination

## Pager
Provides quick previous and next links. `centers` by default

<Playground>
  <Pager>
    <Pager.Item href="#">Previous</Pager.Item>{' '}
    <Pager.Item href="#">Next</Pager.Item>
  </Pager>
</Playground>

## Aligned
Set the `previous` or `next` prop to true, to align left or right.
<Playground>
  <Pager>
    <Pager.Item previous href="#">
      &larr; Previous Page
    </Pager.Item>
    <Pager.Item next href="#">
      Next Page &rarr;
    </Pager.Item>
  </Pager>
</Playground>

## Disabled

Set the `disabled` prop to true to disable the link.

<Playground>
  <Pager>
    <Pager.Item previous href="#">
      &larr; Previous
    </Pager.Item>
    <Pager.Item disabled next href="#">
      Next &rarr;
    </Pager.Item>
  </Pager>
</Playground>

## Pagination
A set of presentational components for building pagination UI.

<Playground>
  {() => {
    let active = 7;
    let items = [];
    for (let number = 1; number <= 10; number++) {
      items.push(
        <Pagination.Item key={number + '-key'} active={number === active}>{number}</Pagination.Item>
      );
    }

    const paginationBasic = (
      <div>
        <Pagination bsSize="large">{items}</Pagination>
        <br />

        <Pagination bsSize="medium">{items}</Pagination>
        <br />

        <Pagination bsSize="small">{items}</Pagination>
      </div>
    );

    return (paginationBasic);
  }}
</Playground>

## Pagination sub-components
For building more complex pagination UI, there are few convenient sub-components
for adding `First`, `Previous`, `Next`, `Last` buttons, as well as an `Ellipsis` item for indicating previous or continuing results.

<Playground>
  <Pagination>
    <Pagination.First />
    <Pagination.Prev />
    <Pagination.Item>{1}</Pagination.Item>
    <Pagination.Ellipsis />

    <Pagination.Item>{10}</Pagination.Item>
    <Pagination.Item>{11}</Pagination.Item>
    <Pagination.Item active>{12}</Pagination.Item>
    <Pagination.Item>{13}</Pagination.Item>
    <Pagination.Item disabled>{14}</Pagination.Item>

    <Pagination.Ellipsis />
    <Pagination.Item>{20}</Pagination.Item>
    <Pagination.Next />
    <Pagination.Last />
  </Pagination>
</Playground>
